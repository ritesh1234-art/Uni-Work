/* Cross-platform getch() - uses conio.h on Windows, termios.h on Linux/Mac */
#ifdef _WIN32
    #include <conio.h>
    #define _getch _getch
    #define CLEAR "cls"
#else
    #include <termios.h>
    #include <unistd.h>
    #define CLEAR "clear"
    char _getch() {
        struct termios oldt, newt;
        char ch;
        tcgetattr(STDIN_FILENO, &oldt);
        newt = oldt;
        newt.c_lflag &= ~(ICANON | ECHO);
        tcsetattr(STDIN_FILENO, TCSANOW, &newt);
        ch = getchar();
        tcsetattr(STDIN_FILENO, TCSANOW, &oldt);
        return ch;
    }
#endif

/* ===================== WORDLE GAME ===================== */
void wordle_game() {
    const char *words[] = {"apple", "grape", "mango", "peach", "berry", "lemon", "melon", "chili", "bread", "cheese"};
    int word_count = 10;
    srand(time(NULL));
    const char *secret = words[rand() % word_count];
    char guess[6];
    int attempts = 6;
    int won = 0;

    system(CLEAR);
    printf(BG_GREEN BOLD " WORDLE TERMINAL " RESET "\n\n");
    printf(GREEN "Green" RESET " = Correct | " YELLOW "Yellow" RESET " = Wrong Spot | " BG_GRAY "Gray" RESET " = Miss\n\n");

    while (attempts > 0) {
        printf(CYAN "Guesses left [%d]> " RESET, attempts);
        scanf("%5s", guess);

        if (strlen(guess) != 5) {
            printf(RED "Error: Must be exactly 5 letters.\n\n" RESET);
            continue;
        }

        if (strcmp(guess, secret) == 0) {
            printf("\n  " BG_GREEN BOLD " %s " RESET "\n\n", guess);
            printf(GREEN "WINNER! You cracked the system! 🎉\n" RESET);
            won = 1;
            break;
        }

        int used[5] = {0};
        char feedback[5] = {'W','W','W','W','W'}; 

        for (int i = 0; i < 5; i++) {
            if (guess[i] == secret[i]) {
                feedback[i] = 'G';
                used[i] = 1;
            }
        }

        for (int i = 0; i < 5; i++) {
            if (feedback[i] == 'G') continue;
            for (int j = 0; j < 5; j++) {
                if (!used[j] && guess[i] == secret[j]) {
                    feedback[i] = 'Y';
                    used[j] = 1;
                    break;
                }
            }
        }

        printf("  ");
        for (int i = 0; i < 5; i++) {
            if(feedback[i] == 'G') printf(BG_GREEN BOLD " %c " RESET " ", guess[i]);
            else if(feedback[i] == 'Y') printf(BG_YELLOW BOLD " %c " RESET " ", guess[i]);
            else printf(BG_GRAY BOLD " %c " RESET " ", guess[i]);
        }
        printf("\n\n");

        attempts--;
    }

    if (!won)
        printf(RED "SYSTEM LOCKED. The word was '%s'\n" RESET, secret);
        
    printf("\nPress any key to return to Main Menu...");
    _getch();
}

/* ===================== TIC TAC TOE ===================== */
int check_winner(char board[]) {
    int win[8][3] = {{0,1,2},{3,4,5},{6,7,8},{0,3,6},{1,4,7},{2,5,8},{0,4,8},{2,4,6}};
    for (int i = 0; i < 8; i++) {
        if (board[win[i][0]] == board[win[i][1]] && board[win[i][1]] == board[win[i][2]]) {
            if (board[win[i][0]] == 'X') return -10;
            if (board[win[i][0]] == 'O') return 10;
        }
    }
    return 0;
}

int is_moves_left(char board[]) {
    for (int i = 0; i < 9; i++) if (board[i] != 'X' && board[i] != 'O') return 1;
    return 0;
}

int minimax(char board[], int depth, int isMax) {
    int score = check_winner(board);
    if (score == 10) return score - depth;
    if (score == -10) return score + depth;
    if (!is_moves_left(board)) return 0;

    if (isMax) {
        int best = -1000;
        for (int i = 0; i < 9; i++) {
            if (board[i] != 'X' && board[i] != 'O') {
                char temp = board[i]; board[i] = 'O';
                int val = minimax(board, depth + 1, 0);
                best = (best > val) ? best : val; board[i] = temp;
            }
        }
        return best;
    } else {
        int best = 1000;
        for (int i = 0; i < 9; i++) {
            if (board[i] != 'X' && board[i] != 'O') {
                char temp = board[i]; board[i] = 'X';
                int val = minimax(board, depth + 1, 1);
                best = (best < val) ? best : val; board[i] = temp;
            }
        }
        return best;
    }
}

int best_move(char board[]) {
    int bestVal = -1000; int move = -1;
    for (int i = 0; i < 9; i++) {
        if (board[i] != 'X' && board[i] != 'O') {
            char temp = board[i]; board[i] = 'O';
            int moveVal = minimax(board, 0, 0); board[i] = temp;
            if (moveVal > bestVal) { bestVal = moveVal; move = i; }
        }
    }
    return move;
}

void print_board(char board[]) {
    printf("\n");
    for (int i = 0; i < 9; i++) {
        if(board[i] == 'X') printf(CYAN BOLD " %c " RESET, board[i]);
        else if(board[i] == 'O') printf(RED BOLD " %c " RESET, board[i]);
        else printf(" %c ", board[i]);
        
        if (i % 3 != 2) printf(YELLOW "|" RESET);
        if (i == 2 || i == 5) printf(YELLOW "\n-----------\n" RESET);
    }
    printf("\n\n");
}

void tic_tac_toe() {
    char board[9] = {'1','2','3','4','5','6','7','8','9'};
    int move;
    
    system(CLEAR);
    printf(BG_GREEN BOLD " TIC TAC TOE: CPU BATTLE " RESET "\n");
    printf(CYAN "You are X" RESET " | " RED "CPU is O\n" RESET);

    for (int turn = 0; turn < 9; turn++) {
        print_board(board);

        if (turn % 2 == 0) {
            printf(CYAN "Your move: " RESET);
            scanf("%d", &move);
            move--;

            if (move < 0 || move > 8 || board[move] == 'X' || board[move] == 'O') {
                printf(RED "Invalid cell override!\n" RESET);
                turn--; continue;
            }
            board[move] = 'X';
        } else {
            printf(MAGENTA "CPU algorithm computing...\n" RESET);
            move = best_move(board); board[move] = 'O';
        }

        int result = check_winner(board);
        if (result == 10) { print_board(board); printf(RED "FATAL ERROR: CPU wins 🤖\n" RESET); break; }
        if (result == -10) { print_board(board); printf(GREEN "VICTORY: You beat the algorithm 🎉\n" RESET); break; }
    }
    if(check_winner(board) == 0) { print_board(board); printf(YELLOW "STALEMATE It's a draw 🤝\n" RESET); }
    
    printf("\nPress any key to return to Main Menu..."); _getch();
}

/* ===================== MATH SURVIVAL GAME ===================== */
void math_game() {
    int level = 1, score = 0;
    const int TIME_LIMIT = 5;
    srand(time(NULL));

    system(CLEAR);
    printf(BG_GREEN BOLD " MATH SURVIVAL PROTOCOL " RESET "\n\n");
    printf(YELLOW "Time Limit: %ds per cycle.\n" RESET, TIME_LIMIT);
    printf(RED "1 mistake = IMMEDIATE TERMINATION.\n\n" RESET);

    while (1) {
        int max_num, correct, user; char op; time_t start, end;
        if (level <= 3) { max_num = 10; op = (rand() % 2 == 0) ? '+' : '-'; } 
        else if (level <= 6) { max_num = 20; int r = rand() % 3; op = (r == 0) ? '+' : (r == 1) ? '-' : '*'; } 
        else { max_num = 50; int r = rand() % 4; op = (r == 0) ? '+' : (r == 1) ? '-' : (r == 2) ? '*' : '/'; }

        int a = rand() % max_num + 1, b = rand() % max_num + 1;
        if (op == '/') { correct = a; a = a * b; } 
        else if (op == '+') correct = a + b; 
        else if (op == '-') correct = a - b; 
        else correct = a * b;

        printf(MAGENTA "LEVEL %d " RESET "| Score %d\n", level, score);
        printf(CYAN "Solve: %d %c %d = " RESET, a, op, b);

        start = time(NULL); scanf("%d", &user); end = time(NULL);

        if (difftime(end, start) > TIME_LIMIT) { printf(RED "\n⏱️ TIMEOUT! Core breached.\n" RESET); break; }
        if (user != correct) { printf(RED "\n❌ MISCALCULATION! Correct value: %d\n" RESET, correct); break; }

        printf(GREEN "✅ Bypass accepted.\n\n" RESET);
        score++; level++;
    }

    printf(BG_RED BOLD "\n === PROTOCOL FAILED === " RESET "\n");
    printf("Final Rating - Level: %d | Score: %d\n\n", level, score);
    printf("Press any key to return to Main Menu..."); _getch();
}

    system(CLEAR);
    printf(CYAN BOLD "\n MAZE RUNNER 2: THE DOOMSDAY - %s\n" RESET, maze_names[current_maze_index]);
    printf(MAGENTA " ==========================================\n" RESET);
    printf(YELLOW " Moves: %03d       [W] Up [S] Down [A] Left [D] Right\n" RESET, moves);
    printf(MAGENTA " ------------------------------------------\n\n" RESET);
    for (int r = 0; r < ROWS; r++) {
        printf("  ");
        for (int c = 0; c < COLS; c++) {
            if (r == player_row && c == player_col) printf(YELLOW BOLD "■ " RESET); // Player
            else {
                switch (current_maze[r][c]) {
                    case WALL:  printf(BLUE "▓▓" RESET); break; // Wall Blocks
                    case PATH:  printf("  "); break;  
                    case START: printf("  "); break;  
                    case EXIT:  printf(BG_GREEN BOLD "  " RESET); break; // Exit door
                    default:    printf("  "); break;
                }
            }
        }
        printf("\n");
    }
    printf(MAGENTA "\n ------------------------------------------\n" RESET);
    printf(" Legend: " YELLOW "■" RESET "=You " BLUE "▓" RESET "=Wall " BG_GREEN "  " RESET "=Exit | " RED "[Q] Quit [R] Restart\n\n" RESET);
}

    system(CLEAR);

    printf(GREEN "\n Maze  : %s\n Moves : %d\n" RESET, maze_names[current_maze_index], moves);

    system(CLEAR);
    printf(BG_RED BOLD "\n\n  CRITICAL IMPACT - PLAYER DESTROYED  \n" RESET);
    printf(RED "\n Maze  : %s\n Moves : %d\n" RESET, maze_names[current_maze_index], moves);
    printf("\n Press any key to return to maze deck...");
}

int move_player(int dr, int dc) {
    int nr = player_row + dr; int nc = player_col + dc;
    if (nr < 0 || nr >= ROWS || nc < 0 || nc >= COLS) return 0;
    if (current_maze[nr][nc] == WALL) return 2;   
    player_row = nr; player_col = nc; moves++;
    if (current_maze[nr][nc] == EXIT) return 1;
    return 0;
}

int show_maze_menu() {
    system(CLEAR);
    printf(BG_BLUE BOLD "\n MAZE DECK TERMINAL \n\n" RESET);
    printf(CYAN " [1] " RESET "The Zigzag         " GREEN "(Easy)\n" RESET);
    printf(CYAN " [2] " RESET "The Labyrinth      " YELLOW "(Medium)\n" RESET);
    printf(CYAN " [3] " RESET "The River          " YELLOW "(Medium)\n" RESET);
    printf(CYAN " [4] " RESET "The Chamber        " RED "(Hard)\n" RESET);
    printf(CYAN " [5] " RESET "The Final Gauntlet " MAGENTA "(Very Hard)\n\n" RESET);
    printf(RED " [Q] " RESET "Eject back to Main Menu\n\n");
    printf(CYAN " Select Sequence: " RESET);
    
    char ch = _getch();
    if (ch == 'q' || ch == 'Q') return -1;
    if (ch >= '1' && ch <= '5') return ch - '1';
    return show_maze_menu(); 
}

/* ===================== MAIN MAZE GAME LOOP ===================== */
void maze_game() {
    char input; int result; 
    while (1) {
        current_maze_index = show_maze_menu();
        if (current_maze_index == -1) break;
        
        load_maze(current_maze_index);
        while (1) {
            draw_maze();
            input = _getch(); result = 0; 
            if      (input == 'w' || input == 'W') result = move_player(-1,  0);
            else if (input == 's' || input == 'S') result = move_player( 1,  0);
            else if (input == 'a' || input == 'A') result = move_player( 0, -1);
            else if (input == 'd' || input == 'D') result = move_player( 0,  1);
            else if (input == 'r' || input == 'R') { load_maze(current_maze_index); continue; }
            else if (input == 'q' || input == 'Q') break;

            if (result == 1) { show_win_screen(); _getch(); break; }
            else if (result == 2) { show_game_over(); _getch(); break; }
        }
    }
}
